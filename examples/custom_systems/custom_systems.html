46template<typename Q, typename U>
47concept castable_to = Quantity<Q> && Unit<U> && requires(Q q) { quantity_cast<U>(q); };
48
49void conversions()
50{
51  // fps::yard is not defined in terms of SI units (or vice-versa)
52  // so the conversion between FPS and SI is not possible
53  static_assert(!castable_to<fps::length<fps::yard>, si::kilometre>);
54
55  // si::fps::yard is defined in terms of SI units
56  // so the conversion between FPS and SI is possible
57  static_assert(castable_to<si::fps::length<si::fps::yard>, si::kilometre>);
58  constexpr auto si_fps_yard = si::fps::length<si::fps::yard>(1.);
59  std::cout << quantity_cast<si::kilometre>(si_fps_yard) << "\n";
60}

62void unknown_dimensions()
63{
64  constexpr auto fps_yard = fps::length<fps::yard>(1.);
65  constexpr auto fps_area = fps_yard * fps_yard;
66  std::cout << fps_yard << "\n";
67  std::cout << quantity_cast<decltype(fps_area)::dimension::coherent_unit>(fps_area) << "\n";
68
69  constexpr auto si_fps_yard = si::fps::length<si::fps::yard>(1.);
70  constexpr auto si_fps_area = si_fps_yard * si_fps_yard;
71  std::cout << si_fps_yard << "\n";
72  std::cout << quantity_cast<decltype(si_fps_area)::dimension::coherent_unit>(si_fps_area) << "\n";
73}

77template<Unit U>
78std::ostream& operator<<(std::ostream& os, const U& u)
79{
80  using unit_type = std::remove_cvref_t<decltype(u)>;
81  return os << as_ratio(unit_type::mag) << " x " << unit_type::reference::symbol.standard();
82}
