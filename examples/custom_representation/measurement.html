 10namespace {
 11
 12template<class T>
 13class measurement {
 14public:
 15  using value_type = T;
 16
 17  measurement() = default;
 18
 19  constexpr explicit measurement(const value_type& val, const value_type& err = {}) : value_(val)
 20  {
 21    // it sucks that using declaration cannot be provided for a constructor initializer list
 22    using namespace std;
 23    uncertainty_ = abs(err);
 24  }
 25
 26  constexpr const value_type& value() const { return value_; }
 27  constexpr const value_type& uncertainty() const { return uncertainty_; }
 28
 29  constexpr value_type relative_uncertainty() const { return uncertainty() / value(); }
 30  constexpr value_type lower_bound() const { return value() - uncertainty(); }
 31  constexpr value_type upper_bound() const { return value() + uncertainty(); }
 32
 33  [[nodiscard]] constexpr measurement operator-() const { return measurement(-value(), uncertainty()); }
 34
 35  [[nodiscard]] friend constexpr measurement operator+(const measurement& lhs, const measurement& rhs)
 36  {
 37    using namespace std;
 38    return measurement(lhs.value() + rhs.value(), hypot(lhs.uncertainty(), rhs.uncertainty()));
 39  }
 40
 41  [[nodiscard]] friend constexpr measurement operator-(const measurement& lhs, const measurement& rhs)
 42  {
 43    using namespace std;
 44    return measurement(lhs.value() - rhs.value(), hypot(lhs.uncertainty(), rhs.uncertainty()));
 45  }
 46
 47  [[nodiscard]] friend constexpr measurement operator*(const measurement& lhs, const measurement& rhs)
 48  {
 49    const auto val = lhs.value() * rhs.value();
 50    using namespace std;
 51    return measurement(val, val * hypot(lhs.relative_uncertainty(), rhs.relative_uncertainty()));
 52  }
 53
 54  [[nodiscard]] friend constexpr measurement operator*(const measurement& lhs, const value_type& value)
 55  {
 56    const auto val = lhs.value() * value;
 57    return measurement(val, val * lhs.relative_uncertainty());
 58  }
 59
 60  [[nodiscard]] friend constexpr measurement operator*(const value_type& value, const measurement& rhs)
 61  {
 62    const auto val = rhs.value() * value;
 63    return measurement(val, val * rhs.relative_uncertainty());
 64  }
 65
 66  [[nodiscard]] friend constexpr measurement operator/(const measurement& lhs, const measurement& rhs)
 67  {
 68    const auto val = lhs.value() / rhs.value();
 69    using namespace std;
 70    return measurement(val, val * hypot(lhs.relative_uncertainty(), rhs.relative_uncertainty()));
 71  }
 72
 73  [[nodiscard]] friend constexpr measurement operator/(const measurement& lhs, const value_type& value)
 74  {
 75    const auto val = lhs.value() / value;
 76    return measurement(val, val * lhs.relative_uncertainty());
 77  }
 78
 79  [[nodiscard]] friend constexpr measurement operator/(const value_type& value, const measurement& rhs)
 80  {
 81    const auto val = value / rhs.value();
 82    return measurement(val, val * rhs.relative_uncertainty());
 83  }
 84
 85  [[nodiscard]] constexpr auto operator<=>(const measurement&) const = default;
 86
 87  friend std::ostream& operator<<(std::ostream& os, const measurement& v)
 88  {
 89    return os << v.value() << " ± " << v.uncertainty();
 90  }
 91
 92private:
 93  value_type value_{};
 94  value_type uncertainty_{};
 95};
 96
 97}  // namespace

104void example()

106  using namespace units::isq;
107
108  const auto a = si::acceleration<si::metre_per_second_sq, measurement<double>>(measurement(9.8, 0.1));
109  const auto t = si::time<si::second, measurement<double>>(measurement(1.2, 0.1));
110
111  const Speed auto v1 = a * t;
112#if UNITS_DOWNCAST_MODE == 0
113  std::cout << a << " * " << t << " = " << v1 << " = " << quantity_cast<si::dim_speed, si::kilometre_per_hour>(v1)
114            << '\n';
115#else
116  std::cout << a << " * " << t << " = " << v1 << " = " << quantity_cast<si::kilometre_per_hour>(v1) << '\n';
117#endif
118
119  si::length<si::metre, measurement<double>> length(measurement(123., 1.));
120  std::cout << "10 * " << length << " = " << 10 * length << '\n';
121}
