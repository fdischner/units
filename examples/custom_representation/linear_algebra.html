 12template<class ET, class OT>
 13std::ostream& operator<<(std::ostream& os, const vector<ET, OT>& v)
 14{
 15  os << "|";
 16  for (auto i = 0U; i < v.size(); ++i) {
 17    os << STD_FMT::format(" {:>9}", v(i));
 18  }
 19  os << " |";
 20  return os;
 21}
 22
 23template<class ET, class OT>
 24std::ostream& operator<<(std::ostream& os, const matrix<ET, OT>& v)
 25{
 26  for (auto i = 0U; i < v.rows(); ++i) {
 27    os << "|";
 28    for (auto j = 0U; j < v.columns(); ++j) {
 29      os << STD_FMT::format(" {:>9}", v(i, j));
 30    }
 31    os << (i != v.rows() - 1U ? " |\n" : " |");
 32  }
 33  return os;
 34}
