 1#include <units/format.h>
 2#include <units/isq/si/length.h>
 3#include <iostream>
 4#include <type_traits>
 5
 6/*
 7  get conversion factor from one dimensionally equivalent
 8  quantity type to another
 9*/
10
11namespace {
12
13template<units::Quantity Target, units::Quantity Source>
14  requires units::equivalent<typename Source::dimension, typename Target::dimension>
15inline constexpr std::common_type_t<typename Target::rep, typename Source::rep> conversion_factor(Target, Source)
16{
17  // get quantities looking like inputs but with Q::rep that doesn't have narrowing conversion
18  typedef std::common_type_t<typename Target::rep, typename Source::rep> rep;
19  typedef units::quantity<typename Source::dimension, typename Source::unit, rep> source;
20  typedef units::quantity<typename Target::dimension, typename Target::unit, rep> target;
21  return target{source{1}}.number();
22}
23
24}  // namespace
25
26int main()
27{
28  using namespace units::isq::si;
29
30  std::cout << "conversion factor in mp-units...\n\n";
31
32  constexpr length<metre> lengthA(2.0);
33  constexpr length<millimetre> lengthB = lengthA;
34
35  std::cout << STD_FMT::format("lengthA( {} ) and lengthB( {} )\n", lengthA, lengthB)
36            << "represent the same length in different units.\n\n";
37
38  std::cout << STD_FMT::format("therefore ratio lengthA / lengthB == {}\n\n", lengthA / lengthB);
39
40  std::cout << STD_FMT::format("conversion factor from lengthA::unit of {:%q} to lengthB::unit of {:%q}:\n\n", lengthA,
41                               lengthB)
42            << STD_FMT::format("lengthB.number( {} ) == lengthA.number( {} ) * conversion_factor( {} )\n",
43                               lengthB.number(), lengthA.number(), conversion_factor(lengthB, lengthA));
44}
