 28// state
 29template<QuantityOrQuantityPoint... QQPs>
 30  requires(sizeof...(QQPs) > 0) && (sizeof...(QQPs) <= 3) && are_derivatives<QQPs...>
 31struct state {
 32  std::tuple<QQPs...> variables_;
 33  constexpr state(QQPs... qqps) : variables_(std::move(qqps)...) {}
 34};

 36template<typename T>
 37concept State = units::is_specialization_of<T, state>;

 39template<std::size_t Idx, typename... Qs>
 40constexpr auto& get(state<Qs...>& s)
 41{
 42  return get<Idx>(s.variables_);
 43}

 45template<std::size_t Idx, typename... Qs>
 46constexpr const auto& get(const state<Qs...>& s)
 47{
 48  return get<Idx>(s.variables_);

 70// kalman gain
 71template<units::Quantity Q>
 72constexpr units::dimensionless<units::one> kalman_gain(Q estimate_uncertainty, Q measurement_uncertainty)
 73{
 74  return estimate_uncertainty / (estimate_uncertainty + measurement_uncertainty);
 75}

 77// state update
 78template<typename Q, QuantityOrQuantityPoint QM, units::Dimensionless K>
 79  requires units::equivalent<typename Q::dimension, typename QM::dimension>
 80constexpr state<Q> state_update(const state<Q>& predicted, QM measured, K gain)
 81{
 82  return {get<0>(predicted) + gain * (measured - get<0>(predicted))};
 83}
 84
 85template<typename Q1, typename Q2, QuantityOrQuantityPoint QM, units::Dimensionless K, units::isq::Time T>
 86  requires units::equivalent<typename Q1::dimension, typename QM::dimension>
 87constexpr state<Q1, Q2> state_update(const state<Q1, Q2>& predicted, QM measured, std::array<K, 2> gain, T interval)
 88{
 89  const auto q1 = get<0>(predicted) + get<0>(gain) * (measured - get<0>(predicted));
 90  const auto q2 = get<1>(predicted) + get<1>(gain) * (measured - get<0>(predicted)) / interval;
 91  return {q1, q2};
 92}
 93
 94template<typename Q1, typename Q2, typename Q3, QuantityOrQuantityPoint QM, units::Dimensionless K, units::isq::Time T>
 95  requires units::equivalent<typename Q1::dimension, typename QM::dimension>
 96constexpr state<Q1, Q2, Q3> state_update(const state<Q1, Q2, Q3>& predicted, QM measured, std::array<K, 3> gain,
 97                                         T interval)
 98{
 99  const auto q1 = get<0>(predicted) + get<0>(gain) * (measured - get<0>(predicted));
100  const auto q2 = get<1>(predicted) + get<1>(gain) * (measured - get<0>(predicted)) / interval;
101  const auto q3 = get<2>(predicted) + get<2>(gain) * (measured - get<0>(predicted)) / (interval * interval / 2);
102  return {q1, q2, q3};
103}

105// covariance update
106template<units::Quantity Q, units::Dimensionless K>
107constexpr Q covariance_update(Q uncertainty, K gain)
108{
109  return (1 - gain) * uncertainty;

112// state extrapolation
113template<typename Q1, typename Q2, units::isq::Time T>
114constexpr state<Q1, Q2> state_extrapolation(const state<Q1, Q2>& estimated, T interval)
115{
116  const auto q1 = get<0>(estimated) + get<1>(estimated) * interval;
117  const auto q2 = get<1>(estimated);
118  return {q1, q2};
119}

121template<typename Q1, typename Q2, typename Q3, units::isq::Time T>
122constexpr state<Q1, Q2, Q3> state_extrapolation(const state<Q1, Q2, Q3>& estimated, T interval)
123{
124  const auto q1 = get<0>(estimated) + get<1>(estimated) * interval + get<2>(estimated) * pow<2>(interval) / 2;
125  const auto q2 = get<1>(estimated) + get<2>(estimated) * interval;
126  const auto q3 = get<2>(estimated);
127  return {q1, q2, q3};
128}

139template<typename... Qs>
140struct STD_FMT::formatter<kalman::state<Qs...>> {
141  constexpr auto parse(format_parse_context& ctx)
142  {
143    units::detail::dynamic_specs_handler handler(specs, ctx);
144    return units::detail::parse_format_specs(ctx.begin(), ctx.end(), handler);
145  }
146
147  template<typename FormatContext>
148  auto format(const kalman::state<Qs...>& s, FormatContext& ctx)
149  {
150    std::string value_buffer;
151    auto to_value_buffer = std::back_inserter(value_buffer);
152    if (specs.precision != -1) {
153      if constexpr (sizeof...(Qs) == 1)
154        STD_FMT::format_to(to_value_buffer, "{1:%.{0}Q %q}", specs.precision, kalman::get<0>(s));
155      else if constexpr (sizeof...(Qs) == 2)
156        STD_FMT::format_to(to_value_buffer, "{{ {1:%.{0}Q %q}, {2:%.{0}Q %q} }}", specs.precision, kalman::get<0>(s),
157                           kalman::get<1>(s));
158      else
159        STD_FMT::format_to(to_value_buffer, "{{ {1:%.{0}Q %q}, {2:%.{0}Q %q}, {3:%.{0}Q %q} }}", specs.precision,
160                           kalman::get<0>(s), kalman::get<1>(s), kalman::get<2>(s));
161    } else {
162      if constexpr (sizeof...(Qs) == 1)
163        STD_FMT::format_to(to_value_buffer, "{}", kalman::get<0>(s));
164      else if constexpr (sizeof...(Qs) == 2)
165        STD_FMT::format_to(to_value_buffer, "{{ {}, {} }}", kalman::get<0>(s), kalman::get<1>(s));
166      else
167        STD_FMT::format_to(to_value_buffer, "{{ {}, {}, {} }}", kalman::get<0>(s), kalman::get<1>(s),
168                           kalman::get<2>(s));
169    }
170
171    std::string global_format_buffer;
172    units::detail::quantity_global_format_specs<char> global_specs = {specs.fill, specs.align, specs.width};
173    units::detail::format_global_buffer(std::back_inserter(global_format_buffer), global_specs);
174
175    return STD_FMT::vformat_to(ctx.out(), global_format_buffer, STD_FMT::make_format_args(value_buffer));
176  }
177private:
178  units::detail::dynamic_format_specs<char> specs;
179};

181template<typename Q>
182struct STD_FMT::formatter<kalman::estimation<Q>> {
183  constexpr auto parse(format_parse_context& ctx)
184  {
185    units::detail::dynamic_specs_handler handler(specs, ctx);
186    return units::detail::parse_format_specs(ctx.begin(), ctx.end(), handler);
187  }
188
189  template<typename FormatContext>
190  auto format(kalman::estimation<Q> e, FormatContext& ctx)
191  {
192    units::Quantity auto q = [](const Q& t) {
193      if constexpr (units::Quantity<Q>)
194        return t;
195      else
196        return t.relative();
197    }(kalman::get<0>(e.state));
198
199    std::string value_buffer;
200    auto to_value_buffer = std::back_inserter(value_buffer);
201    if (specs.precision != -1) {
202      STD_FMT::format_to(to_value_buffer, "{0:%.{2}Q} ± {1:%.{2}Q} {0:%q}", q, sqrt(e.uncertainty), specs.precision);
203    } else {
204      STD_FMT::format_to(to_value_buffer, "{0:%Q} ± {1:%Q} {0:%q}", q, sqrt(e.uncertainty));
205    }
206
207    std::string global_format_buffer;
208    units::detail::quantity_global_format_specs<char> global_specs = {specs.fill, specs.align, specs.width};
209    units::detail::format_global_buffer(std::back_inserter(global_format_buffer), global_specs);
210
211    return STD_FMT::vformat_to(ctx.out(), global_format_buffer, STD_FMT::make_format_args(value_buffer));
212  }
213private:
214  units::detail::dynamic_format_specs<char> specs;
215};
