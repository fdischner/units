Example 5 - Estimating the building height

Implementation of: https://www.kalmanfilter.net/kalman1d.html#ex5

kalman_filter-example_5.cpp
 1#include "kalman.h"
 2#include <units/format.h>
 3#include <units/isq/si/length.h>
 4#include <units/math.h>
 5#include <array>
 6#include <iostream>
 7
 8// Based on: https://www.kalmanfilter.net/kalman1d.html#ex5
 9
10using namespace units;
11
12template<Quantity Q>
13void print_header(kalman::estimation<Q> initial)
14{
15  std::cout << STD_FMT::format("Initial: {}\n", initial);
16  std::cout << STD_FMT::format("{:>2} | {:>5} | {:>8} | {:>16} | {:>16}\n", "N", "Gain", "Measured", "Curr. Estimate",
17                               "Next Estimate");
18}
19
20template<Quantity Q, Dimensionless K>
21void print(auto iteration, K gain, Q measured, kalman::estimation<Q> current, kalman::estimation<Q> next)
22{
23  std::cout << STD_FMT::format("{:2} | {:5%.2Q} | {:8} | {:>16.2} | {:>16.2}\n", iteration, gain, measured, current,
24                               next);
25}
26
27int main()
28{
29  using namespace kalman;
30  using namespace units::isq;
31  using namespace units::isq::si::references;
32
33  const estimation initial = {state{60. * m}, pow<2>(15. * m)};
34  const std::array measurements = {48.54 * m, 47.11 * m, 55.01 * m, 55.15 * m, 49.89 * m,
35                                   40.85 * m, 46.72 * m, 50.05 * m, 51.27 * m, 49.95 * m};
36  const auto measurement_uncertainty = pow<2>(5. * m);
37
38  auto update = [=]<Quantity Q>(const estimation<Q>& previous, const Q& meassurement, Dimensionless auto gain) {
39    return estimation{state_update(previous.state, meassurement, gain), covariance_update(previous.uncertainty, gain)};
40  };
41
42  auto predict = []<Quantity Q>(const estimation<Q>& current) { return current; };
43
44  print_header(initial);
45  estimation next = predict(initial);
46  for (int index = 1; const auto& measured : measurements) {
47    const auto& previous = next;
48    const auto gain = kalman_gain(previous.uncertainty, measurement_uncertainty);
49    const estimation current = update(previous, measured, gain);
50    next = predict(current);
51    print(index++, gain, measured, current, next);
52  }
53}
