 43void example()
 44{
 45  // SI (int)
 46  {
 47    using namespace units::isq::si::references;
 48    constexpr auto distance = 220 * km;
 49    constexpr auto duration = 2 * h;
 50
 51    std::cout << "SI units with 'int' as representation\n";
 52
 53    print_result(distance, duration, fixed_int_si_avg_speed(distance, duration));
 54    print_result(distance, duration, fixed_double_si_avg_speed(distance, duration));
 55    print_result(distance, duration, si_avg_speed(distance, duration));
 56    print_result(distance, duration, avg_speed(distance, duration));
 57  }
 58
 59  // SI (double)
 60  {
 61    using namespace units::isq::si::references;
 62    constexpr auto distance = 220. * km;
 63    constexpr auto duration = 2. * h;
 64
 65    std::cout << "\nSI units with 'double' as representation\n";
 66
 67    // conversion from a floating-point to an integral type is a truncating one so an explicit cast is needed
 68    print_result(distance, duration,
 69                 fixed_int_si_avg_speed(quantity_cast<int>(distance), quantity_cast<int>(duration)));
 70
 71    print_result(distance, duration, fixed_double_si_avg_speed(distance, duration));
 72    print_result(distance, duration, si_avg_speed(distance, duration));
 73    print_result(distance, duration, avg_speed(distance, duration));
 74  }
