12#include <units/isq/si/volume.h>
13#include <cassert>
14#include <iostream>
15#include <utility>

17namespace {
18
19using namespace units::isq;
20using namespace si::mass_references;
21using namespace si::volume_references;
22
23inline constexpr Acceleration auto g = si::si2019::standard_gravity<>;
24inline constexpr Density auto air_density = 1.225 * (kg / m3);
25
26class Box {
27  si::area<si::square_metre> base_;
28  si::length<si::metre> height_;
29  si::density<si::kilogram_per_metre_cub> density_ = air_density;
30public:
31  constexpr Box(const si::length<si::metre>& length, const si::length<si::metre>& width, si::length<si::metre> height) :
32      base_(length * width), height_(std::move(height))
33  {
34  }
35
36  [[nodiscard]] constexpr si::force<si::newton> filled_weight() const
37  {
38    const Volume auto volume = base_ * height_;
39    const Mass auto mass = density_ * volume;
40    return mass * g;
41  }
42
43  [[nodiscard]] constexpr si::length<si::metre> fill_level(const si::mass<si::kilogram>& measured_mass) const
44  {
45    return height_ * measured_mass * g / filled_weight();
46  }
47
48  [[nodiscard]] constexpr si::volume<si::cubic_metre> spare_capacity(const si::mass<si::kilogram>& measured_mass) const
49  {
50    return (height_ - fill_level(measured_mass)) * base_;
51  }
52
53  constexpr void set_contents_density(const si::density<si::kilogram_per_metre_cub>& density_in)
54  {
55    assert(density_in > air_density);
56    density_ = density_in;
57  }
58};
59
60}  // namespace
61
62
63int main()
64{
65  using namespace units;
66  using namespace units::isq::si;
67  using namespace length_references;
68  using namespace time_references;
69
70  const auto height = quantity_cast<metre>(200.0 * mm);
71  auto box = Box(1000.0 * mm, 500.0 * mm, height);
72  box.set_contents_density(1000.0 * (kg / m3));
73
74  const auto fill_time = 200.0 * s;      // time since starting fill
75  const auto measured_mass = 20.0 * kg;  // measured mass at fill_time
76
77  const Length auto fill_level = box.fill_level(measured_mass);
78  const Dimensionless auto fill_percent = quantity_cast<percent>(fill_level / height);
79  const Volume auto spare_capacity = box.spare_capacity(measured_mass);
80  const auto input_flow_rate = measured_mass / fill_time;  // unknown dimension
81  const Speed auto float_rise_rate = fill_level / fill_time;
82  const Time auto fill_time_left = (height / fill_level - 1) * fill_time;
83
84  std::cout << "mp-units box example...\n";
85  std::cout << STD_FMT::format("fill height at {} = {} ({} full)\n", fill_time, fill_level, fill_percent);
86  std::cout << STD_FMT::format("spare_capacity at {} = {}\n", fill_time, spare_capacity);
87  std::cout << STD_FMT::format("input flow rate after {} = {}\n", fill_time, input_flow_rate);
88  std::cout << STD_FMT::format("float rise rate = {}\n", float_rise_rate);
89  std::cout << STD_FMT::format("box full E.T.A. at current flow rate = {}\n", fill_time_left);
90}
