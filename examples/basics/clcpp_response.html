 15void simple_quantities()
 16{
 17  using namespace units::isq;
 18  using namespace units::isq::si;
 19  using namespace units::isq::si::references;
 20  using namespace units::isq::si::international::references;
 21
 22  using distance = si::length<si::metre>;
 23  using duration = si::time<si::second>;
 24
 25  UNITS_DIAGNOSTIC_PUSH
 26  UNITS_DIAGNOSTIC_IGNORE_SHADOW
 27  constexpr distance km = 1.0 * references::km;
 28  constexpr distance miles = 1.0 * mi;
 29
 30  constexpr duration sec = 1 * s;
 31  constexpr duration min = 1 * references::min;
 32  constexpr duration hr = 1 * h;
 33  UNITS_DIAGNOSTIC_POP
 34
 35  std::cout << "A physical quantities library can choose the simple\n";
 36  std::cout << "option to provide output using a single type for each base unit:\n\n";
 37  std::cout << km << '\n';
 38  std::cout << miles << '\n';
 39  std::cout << sec << '\n';
 40  std::cout << min << '\n';
 41  std::cout << hr << "\n\n";
 42}

 97void calcs_comparison()
 98{
 99  using namespace units::isq::si;
100  using namespace units::isq::si::references;
101
102  std::cout << "\nA distinct unit for each type is efficient and accurate\n"
103               "when adding two values of the same very big\n"
104               "or very small type:\n\n";
105
106  const length<femtometre, float> L1A = 2.f * fm;
107  const length<femtometre, float> L2A = 3.f * fm;
108  const length<femtometre, float> LrA = L1A + L2A;
109  std::cout << STD_FMT::format("{:%.30Q %q}\n + {:%.30Q %q}\n   = {:%.30Q %q}\n\n", L1A, L2A, LrA);
110
111  std::cout << "The single unit method must convert large\n"
112               "or small values in other units to the base unit.\n"
113               "This is both inefficient and inaccurate\n\n";
114
115  const length<metre, float> L1B = L1A;
116  const length<metre, float> L2B = L2A;
117  const length<metre, float> LrB = L1B + L2B;
118  std::cout << STD_FMT::format("{:%.30Q %q}\n + {:%.30Q %q}\n   = {:%.30Q %q}\n\n", L1B, L2B, LrB);
119
120  std::cout << "In multiplication and division:\n\n";
121
122  const area<square_femtometre, float> ArA = L1A * L2A;
123  std::cout << STD_FMT::format("{:%.30Q %q}\n * {:%.30Q %q}\n   = {:%.30Q %q}\n\n", L1A, L2A, ArA);
124
125  std::cout << "similar problems arise\n\n";
126
127  const area<square_metre, float> ArB = L1B * L2B;
128  std::cout << STD_FMT::format("{:%.30Q %q}\n * {:%.30Q %q}\n   = {:%.30Q %q}\n\n", L1B, L2B, ArB);
129}
