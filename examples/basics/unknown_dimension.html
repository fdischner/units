 8namespace {
 9
10template<units::isq::Length D, units::isq::Time T>
11constexpr units::isq::Speed auto avg_speed(D d, T t)
12{
13  return d / t;
14}
15
16void example()
17{
18  using namespace units::isq;
19  using namespace units::isq::si::references;
20
21  Length auto d1 = 123 * m;
22  Time auto t1 = 10 * s;
23  Speed auto v1 = avg_speed(d1, t1);
24
25  auto temp1 =
26    v1 * (50 * m);  // produces intermediate unknown dimension with 'unknown_coherent_unit' as its 'coherent_unit'
27  Speed auto v2 = temp1 / (100 * m);  // back to known dimensions again
28  Length auto d2 = v2 * (60 * s);
29
30  std::cout << "d1 = " << d1 << '\n';
31  std::cout << "t1 = " << t1 << '\n';
32  std::cout << "v1 = " << v1 << '\n';
33  std::cout << "temp1 = " << temp1 << '\n';
34  std::cout << "v2 = " << v2 << '\n';
35  std::cout << "d2 = " << d2 << '\n';
36}
37
38}  // namespace
39
40int main()
41{
42  try {
43    example();
44  } catch (const std::exception& ex) {
45    std::cerr << "Unhandled std exception caught: " << ex.what() << '\n';
46  } catch (...) {
47    std::cerr << "Unhandled unknown exception caught\n";
48  }
49}
