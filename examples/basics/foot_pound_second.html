 37// Print 'a' in its current units and print its value cast to the units in each of Args
 38template<class... Args, units::Quantity Q>
 39auto fmt_line(const Q a)
 40{
 41  return STD_FMT::format("{:22}", a) + (STD_FMT::format(",{:20}", units::quantity_cast<Args>(a)) + ...);
 42}

 44// Print the ship details in the units as defined in the Ship struct, in other si::imperial units, and in SI
 45void print_details(std::string_view description, const Ship& ship)
 46{
 47  using namespace units::isq::si::fps::references;
 48  const auto waterDensity = 62.4 * (lb / ft3);
 49  std::cout << STD_FMT::format("{}\n", description);
 50  std::cout << STD_FMT::format("{:20} : {}\n", "length",
 51                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.length))
 52            << STD_FMT::format("{:20} : {}\n", "draft",
 53                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.draft))
 54            << STD_FMT::format("{:20} : {}\n", "beam",
 55                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.beam))
 56            << STD_FMT::format("{:20} : {}\n", "mass",
 57                               fmt_line<si::fps::mass<si::fps::long_ton>, si::mass<si::tonne>>(ship.mass))
 58            << STD_FMT::format(
 59                 "{:20} : {}\n", "speed",
 60                 fmt_line<si::speed<si::international::knot>, si::speed<si::kilometre_per_hour>>(ship.speed))
 61            << STD_FMT::format("{:20} : {}\n", "power",
 62                               fmt_line<si::fps::power<si::fps::horse_power>, si::power<si::kilowatt>>(ship.power))
 63            << STD_FMT::format("{:20} : {}\n", "main guns",
 64                               fmt_line<si::fps::length<si::fps::inch>, si::length<si::millimetre>>(ship.mainGuns))
 65            << STD_FMT::format("{:20} : {}\n", "fire shells weighing",
 66                               fmt_line<si::fps::mass<si::fps::long_ton>, si::mass<si::kilogram>>(ship.shellMass))
 67            << STD_FMT::format(
 68                 "{:20} : {}\n", "fire shells at",
 69                 fmt_line<si::fps::speed<si::fps::mile_per_hour>, si::speed<si::kilometre_per_hour>>(ship.shellSpeed))
 70            << STD_FMT::format("{:20} : {}\n", "volume underwater",
 71                               fmt_line<si::volume<si::cubic_metre>, si::volume<si::litre>>(ship.mass / waterDensity));
 72}
